setTimeout 75000
resizeWindow 1016,664
open /connect/?mapConfigLocation=/connect/config/map-mixedLayers3.xml
store //div[@id="legend"] legendControl
store Conservation Map Layer1
store Development Map Layer2
waitForElementPresent ${legendControl}
waitForElementPresent //div[@id="${Layer1}"]
waitForElementPresent //div[@id="${Layer2}"]
assertAttribute ${legendControl}/div[@class="master"][1]/div@id ${Layer1}
assertAttribute ${legendControl}/div[@class="master"][2]/div@id ${Layer2}
open /connect/?mapConfigLocation=/connect/config/map2.xml
waitForElementPresent ${legendControl}
waitForElementPresent //div[@id="${Layer1}"]
waitForElementPresent //div[@id="${Layer2}"]
assertAttribute ${legendControl}/div[@class="master"][1]/div@id ${Layer2}
assertAttribute ${legendControl}/div[@class="master"][2]/div@id ${Layer1}
