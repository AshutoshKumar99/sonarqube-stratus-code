MapControl_Map_Resize
setTimeout 75000
open /connect/?mapConfigLocation=/connect/config/map2.xml
resizeWindow 1000,700
pause 5000
storeExpression //*[@id='myMap'] mapLocator
waitForElementPresent ${mapLocator}
storeElementWidth ${mapLocator} MapWidthOld
storeElementHeight ${mapLocator} MapHeightOld
storeElementPositionTop ${mapLocator} MapTopOld
storeElementPositionLeft ${mapLocator} MapLeftOld
setTimeout 30000
storeExpression ${mapLocator}//img[contains(@src, '/MapTilingService/RESTService/getTile?name=MasterMap&level=8&row=68&col=69&output=png')] tileLocator
waitForImageLoaded ${tileLocator}
storeElementPositionLeft ${tileLocator} tileLeftAbsolute
storeElementPositionTop ${tileLocator} tileTopAbsolute
resizeWindow 800,500
pause 5000
storeElementWidth ${mapLocator} MapWidthNew
storeElementHeight ${mapLocator} MapHeightNew
storeElementPositionTop ${mapLocator} MapTopNew
storeElementPositionLeft ${mapLocator} MapLeftNew
storeElementPositionLeft ${tileLocator} tileLeftNewAbsolute
storeElementPositionTop ${tileLocator} tileTopNewAbsolute
verifyEval ${MapWidthOld}-${MapWidthNew} 200
verifyEval ${MapLeftOld}-${MapLeftNew} 0
verifyEval ${MapTopOld}-${MapTopNew} 0
pause 5000
storeEval Math.abs((${MapWidthOld}-${MapWidthNew})/2) mapWidthDifference
storeEval Math.abs((${MapHeightOld}-${MapHeightNew})/2) mapHeightDifference
assertEval Math.abs(${tileLeftAbsolute}-${tileLeftNewAbsolute})-${mapWidthDifference}<=10 true
assertEval Math.abs(${tileTopAbsolute}-${tileTopNewAbsolute})-${mapHeightDifference}<=10 true
