open /connect/?mapConfigLocation=/connect/config/map2.xml
resizeWindow 1016,664
store //div[@id='locator']//input[1] SearchInputBox
store //div[@id='locator']//button SubmitButton
verifyElementPresent ${SearchInputBox}
store //div[@id='result'] locatorResultsLocator
type ${SearchInputBox} 20 KING HENRY'S ROAD LONDON NW3 3RP
click ${SubmitButton}
waitForElementPresent ${locatorResultsLocator}//table
click ${locatorResultsLocator}//table//tr[1]
storeExpression //img[contains(@src, "/MapTilingService/RESTService/getTile?name=MasterMap&level=8&row=67&col=68&output=png")] tileLocator
waitForElementPresent ${tileLocator} 20000
storeElementPositionLeft ${tileLocator} tileLeft
storeElementPositionTop ${tileLocator} tileTop
store //div[@id='MapMarker'] markerLocator
waitForElementPresent ${markerLocator} 20000
storeElementPositionLeft ${markerLocator} markerLeft
storeElementPositionTop ${markerLocator} markerTop
store ${markerLeft} - ${tileLeft} differenceLeft
store ${markerTop} - ${tileTop} differenceTop
assertEval ${differenceLeft} regexp:211|221|231
assertEval ${differenceTop} regexp:185
