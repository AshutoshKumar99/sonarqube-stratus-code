Toggle between Layers
store //div[@id="Conservation Map"]/../div[2]/input[@type='checkbox'] CamdenMapCheckbox
assertChecked ${CamdenMapCheckbox}
store //div[@id='OpenLayers.Layer.TileServer_10']/div[1]/img CamdenMapLayer
waitForAttribute ${CamdenMapLayer}@class olTileImage
waitForCondition var img = selenium.browserbot.locateElementByXPath("${CamdenMapLayer}", window.document); img && img.complete && (typeof img.naturalWidth == 'undefined' || img.naturalWidth > 0) 20000
click ${CamdenMapCheckbox}
pause 3000
verifyNotVisible ${CamdenMapLayer}
store //div[@id="Development Map"]/../div[2]/input[@type='checkbox'] PlanningMapCheckbox

waitForCondition var img = selenium.browserbot.locateElementByXPath("${PlanningMapLayer}", window.document); img && img.complete && (typeof img.naturalWidth == 'undefined' || img.naturalWidth > 0) 20000
click ${PlanningMapCheckbox}
pause 3000
verifyNotVisible ${PlanningMapLayer}
