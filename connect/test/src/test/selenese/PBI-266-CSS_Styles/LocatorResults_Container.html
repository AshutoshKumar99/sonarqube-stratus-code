LocatorResults_Container
open /connect/
store //div[@id='locator']//input[1] SearchInputBox
store //div[@id='locator']//button SubmitButton
waitForElementPresent ${SearchInputBox}
storeExpression //div[@id="result"] locatorResultsLocator
storeElementHeight ${locatorResultsLocator} ResultHeight
assertEval ${ResultHeight} 0
type ${SearchInputBox} +1 +erskine +road -2 -floor -flat
click ${SubmitButton}
pause 2000
waitForElementPresent ${locatorResultsLocator}/table
waitForElementPresent ${locatorResultsLocator}/table//tr[@class='normalListColor']
waitForText ${locatorResultsLocator}/table 1 Erskine Road Chalk Farm London NW3 3AJ
pause 5000
verifyTextPresent 1 Erskine Road Chalk Farm London NW3 3AJ
storeElementHeight ${locatorResultsLocator} ExpandedResultHeight
assertEval ${ExpandedResultHeight} 245
assertEval var l = selenium.browserbot.locateElementByXPath('${locatorResultsLocator}', window.document); l.scrollHeight <= l.offsetHeight true
