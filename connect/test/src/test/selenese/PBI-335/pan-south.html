Pan South
setTimeout 75000
open /connect
pause 500
store //div[@id='stratus_PanControl_0']/button[contains(@class, 'south')] south
storeExpression //div[@id='myMap'] mapLocator
waitForElementPresent ${mapLocator}
storeElementHeight ${mapLocator} mapHeight
storeEval ${mapHeight} / 4 panAmountVertical
waitForElementPresent //div[contains(@class, 'panControl')]
storeExpression //img[contains(@src, '/MapTiling/RESTService/getTile?name=StreetProMM&level=8&row=68&col=69&output=png')] tileLocator
waitForImageLoaded ${tileLocator}
storeElementPositionTop ${tileLocator} tileTop
mouseDown ${south}
click ${south}
mouseUp ${south}
pause 5000
storeElementPositionTop ${tileLocator} tileTopNew
verifyEval Math.abs(${panAmountVertical} - (${tileTop} - ${tileTopNew})) < 10 true
